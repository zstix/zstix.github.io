url="http://zackstickles.com"
file="bf.jpg"

dir="$HOME/.tmp"
remote="$url/$file"
local="$dir/$file"

if [ ! -d $dir ]; then
  mkdir $dir
fi

if [ ! -f $local ]; then
  curl -s $remote -o $local
fi

cmd="tell application \"Finder\" to set desktop picture to \""$local"\" as POSIX file"
osascript -e "$cmd"
